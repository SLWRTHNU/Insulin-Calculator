  <script>
    // Shorthand query selector
    function $(sel) { return document.querySelector(sel); }

    function esc(str) {
      return String(str || '')
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#39;');
    }

    // Highlight all occurrences of the typed text (substring, case-insensitive)
    function hi(s, q) {
      var text = String(s || '');
      var needle = String(q || '').trim().toLowerCase();
      if (!needle) return esc(text);

      var lower = text.toLowerCase();
      var result = '';
      var i = 0;

      while (true) {
        var idx = lower.indexOf(needle, i);
        if (idx === -1) {
          result += esc(text.slice(i));
          break;
        }
        // text before the match
        result += esc(text.slice(i, idx));
        // the matched part
        result += '<mark>' + esc(text.slice(idx, idx + needle.length)) + '</mark>';
        i = idx + needle.length;
      }

      return result;
    }

    function render(res, q) {
      var box = $('#results');
      var warn = $('#warn');
      var count = $('#count');

      warn.textContent = res.msg || '';
      count.textContent = '';

      var list = res.rows || [];

      if (res.msg) {
        box.innerHTML = '';
        return;
      }

      if (!list.length) {
        box.innerHTML = '<div class="empty">No matches found.</div>';
        return;
      }

      count.textContent = 'Showing ' + list.length + ' result(s).';

      box.innerHTML = list.map(function(r) {
        return (
          '<div class="row">' +
            '<div class="date">' + hi(r.date, q) + '</div>' +
            '<div class="field"><span class="label">Meal:</span> ' + hi(r.meal, q) + '</div>' +
            '<div class="field"><span class="label">Food:</span> ' + hi(r.food, q) + '</div>' +
            '<div class="field"><span class="label">Carb Factor:</span> ' + hi(r.cf, q) + '</div>' +
            '<div class="field"><span class="label">Weight Given:</span> ' + hi(r.wt, q) + '</div>' +
            '<div class="field"><span class="label">Net Carbs:</span> ' + hi(r.carbs, q) + '</div>' +
            '<div class="link">' +
              '<a href="' + r.docUrl + '" target="_blank">View \'' + esc(r.date) + '\'</a>' +
            '</div>' +
          '</div>'
        );
      }).join('');
    }

    function search() {
      var q = $('#q').value.trim();
      if (!q) {
        $('#results').innerHTML = '';
        $('#warn').textContent = '';
        $('#count').textContent = '';
        return;
      }

      google.script.run
        .withSuccessHandler(function(res) { render(res, q); })
        .withFailureHandler(function(err) {
          $('#warn').textContent = 'Search error: ' + err.message;
        })
        .FoodSearch_query(q);
    }

    function runRebuild(isExport) {
      var btnId = isExport ? '#forceExport' : '#rebuild';
      var otherBtnId = isExport ? '#rebuild' : '#forceExport';
      var serverFn = isExport ? 'FoodSearch_forceExportAndIndex' : 'FoodSearch_rebuildIndex';
      var scanMsg = isExport
        ? 'Exporting today and rebuilding index...'
        : 'Scanning Drive and rebuilding index...';

      $(btnId).disabled = true;
      $(otherBtnId).disabled = true;
      $('#warn').textContent = scanMsg;
      $('#count').textContent = '';
      $('#results').innerHTML = '';

      google.script.run
        .withSuccessHandler(function(info) {
          $(btnId).disabled = false;
          $(otherBtnId).disabled = false;
          $('#warn').textContent = '';

          var count = info && typeof info.count === 'number' ? info.count : 0;
          $('#count').textContent = 'Index has ' + count + ' record(s).';

          // Refresh current search if there is text
          var q = $('#q').value.trim();
          if (q) search();
        })
        .withFailureHandler(function(err) {
          $(btnId).disabled = false;
          $(otherBtnId).disabled = false;
          $('#warn').textContent = 'Rebuild error: ' + err.message;
        })[serverFn]();
    }

    // Event listeners
    $('#q').addEventListener('input', function() {
      clearTimeout(window.__t);
      window.__t = setTimeout(search, 150);
    });
    $('#rebuild').addEventListener('click', function() { runRebuild(false); });
    $('#forceExport').addEventListener('click', function() { runRebuild(true); });

    // Initial index count
    google.script.run
      .withSuccessHandler(function(info) {
        if (info && info.count !== undefined) {
          $('#count').textContent = 'Index has ' + info.count + ' record(s).';
        }
      })
      .withFailureHandler(function() { /* ignore */ })
      .FoodSearch_indexCounts_();
  </script>
