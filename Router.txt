/**
 * Single entry point that delegates to each feature's edit-handler.
 * Keep this as the ONLY onEdit in the project.
 *
 * It is crucial that the "Router" module is the only one containing the
 * special reserved function name `onEdit(e)`.
 */
function onEdit(e) {
  if (!e) return; // Safety check if run manually

  // Call each module's handler, trapping errors so one failure doesn't block others.
  
  // 1. Update Meal Values (calculates carbs/factor/weight based on input)
  // Assumes function is named 'onEdit_UpdateMealValues'
  try { 
    if (typeof onEdit_UpdateMealValues === 'function') {
      onEdit_UpdateMealValues(e); 
    }
  } catch (err) { 
    console.warn("Router Error: onEdit_UpdateMealValues failed.", err.message, err.stack); 
  }
  
  // 2. Post Bolus Tracking (triggers Nightscout lookup after bolus time is logged)
  // Assumes function is named 'onEdit_PostBolusTracking'
  try { 
    if (typeof onEdit_PostBolusTracking === 'function') {
      onEdit_PostBolusTracking(e); 
    }
  } catch (err) { 
    console.warn("Router Error: onEdit_PostBolusTracking failed.", err.message, err.stack); 
  }
}


function onOpen(e) {
  try {
    if (typeof onOpen_FoodSearchMenu === 'function') {
      onOpen_FoodSearchMenu();
    }
  } catch (err) {
    Logger.log('Error building Food Search menu: ' + err);
  }

  try {
    if (typeof onOpen_DailyExportMenu === 'function') {
      onOpen_DailyExportMenu();
    }
  } catch (err) {
    Logger.log('Error building Daily Export menu: ' + err);
  }

  try {
    if (typeof onOpen_MealBuilderMenu === 'function') {
      onOpen_MealBuilderMenu();
    }
  } catch (err) {
    Logger.log('Error building Meal Builder menu: ' + err);
  }
}
